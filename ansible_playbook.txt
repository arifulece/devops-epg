Example playbook
---
- name: Create a Web content directly
  hosts: dev				[for multiple hosts what i do ??]
  tasks:
    - name: Create webdev user
	 user:
	    name: webdev
		state: present
	- name: Install httpd and firewalld
	  yum:
	    name:
		  - httpd
		  - firewalld
		state: present
	- name: Started and Enabled httpd and firewalld
	  service:
	    name: "{{ item }}"
		state: started
		enabled: yes
	  loop:
	    - httpd
		- firewalld
	- name: Allow http in the firewalld
	  firewalld:
	  service: http
	  state: enabled
	  permanent: yes
	  immediate: yes
	- name: Create /webdev directory
	  file:
	    path: /webdev
		group: webdev
		mode: 2775
		state: directory
		setype: httpd_sys_content_t
